// Remove any additional parameters if (fileId && fileId.includes('&')) { fileId = fileId.split('&')[0]; } } else if (driveUrl.includes('/d/')) { // Format: https://drive.google.com/file/d/FILE_ID/view var match = driveUrl.match(/\/d\/([^\/]+)/);
